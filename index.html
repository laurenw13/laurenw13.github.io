def check_nan(input):
    input = input.lstrip().lower()
    first_word = input.split(' ', 1)[0]
    if first_word in ['none','n/a','n\\a','na','nan','no','unknown']:
        return True
    else:
        return False

shuffled = patients.sample(frac=1)
result = np.array_split(shuffled, 2)  

train_set = result[0]
test_set = result[1]

for idx, dataset in enumerate([train_set, test_set]):
    dataset = dataset.groupby(['AgeGroup','SEX','ONSET_DATE'], as_index = False).agg('count')
    dataset = dataset[['AgeGroup','SEX','ONSET_DATE','VAERS_ID']]
    age_group = pd.get_dummies(dataset.AgeGroup, drop_first = True)
    sex = pd.get_dummies(dataset.SEX, drop_first = True)
    standard = dataset[['ONSET_DATE','VAERS_ID']]
    combined = pd.concat([standard, age_group, sex], axis = 1)
    combined['VAERS_ID'] = combined['VAERS_ID'].fillna(0.0)
    combined = combined.rename({'VAERS_ID': 'count'}, axis=1) 
    combined['ONSET_DATE'] = combined['ONSET_DATE'].map(lambda x: x.month)
    
    if idx == 0:
        train_set = combined
    else:
        test_set = combined
